tpauto_wgoa_mod_is_active = {
	# There are no obvious global flags which will always be present in a W:GoA game.
	# However, it has a lot of seemingly-unique game rules, so we can check that several of those are present.
	calc_true_if = {
		amount >= 6 # 6/8 is enough to be pretty sure this is W:GoA, while allowing for typos & future wgoa-mod changes
		OR = {
			has_game_rule = { name = scourge_invasion value = default }
			has_game_rule = { name = scourge_invasion value = x1_5    }
			has_game_rule = { name = scourge_invasion value = x2      }
			has_game_rule = { name = scourge_invasion value = off     }
			has_game_rule = { name = scourge_invasion value = x0_25   }
			has_game_rule = { name = scourge_invasion value = x0_5    }
		}
		OR = {
			has_game_rule = { name = scourge_spawn value = random }
			has_game_rule = { name = scourge_spawn value = always }
		}
		OR = {
			has_game_rule = { name = legion_invasion value = default }
			has_game_rule = { name = legion_invasion value = x1_5    }
			has_game_rule = { name = legion_invasion value = x2      }
			has_game_rule = { name = legion_invasion value = off     }
			has_game_rule = { name = legion_invasion value = x0_25   }
			has_game_rule = { name = legion_invasion value = x0_5    }
		}
		OR = {
			has_game_rule = { name = horde_invasion value = default }
			has_game_rule = { name = horde_invasion value = x1_5    }
			has_game_rule = { name = horde_invasion value = x2      }
			has_game_rule = { name = horde_invasion value = x0_25   }
			has_game_rule = { name = horde_invasion value = x0_5    }
		}
		OR = {
			has_game_rule = { name = old_god_black_empire_invasion value = on      }
			has_game_rule = { name = old_god_black_empire_invasion value = x1_5    }
			has_game_rule = { name = old_god_black_empire_invasion value = x2      }
			has_game_rule = { name = old_god_black_empire_invasion value = off     }
			has_game_rule = { name = old_god_black_empire_invasion value = x0_25   }
			has_game_rule = { name = old_god_black_empire_invasion value = x0_5    }
		}
		OR = {
			has_game_rule = { name = halfbreed_race value = dynasty }
			has_game_rule = { name = halfbreed_race value = father  }
			has_game_rule = { name = halfbreed_race value = mother  }
			has_game_rule = { name = halfbreed_race value = random  }
		}
		OR = {
			has_game_rule = { name = mists_of_pandaria value = on  }
			has_game_rule = { name = mists_of_pandaria value = off }
		}
		OR = {
			has_game_rule = { name = uldum_isolation value = on  }
			has_game_rule = { name = uldum_isolation value = off }
		}
	}
}

		limit = { evil_public_religion_trigger = yes }

	trigger_else = {
		evil_public_religion_trigger = no
	}
	
	# Base mod condition: This character must not be evil or tribal
