edmp_interference_root_is_valid_potential_target = {
	ROOT = {
		is_ruler = yes
		OR = {
			is_landed = yes
			mercenary = no
			edmp_this_character_is_a_current_or_absent_human_player = yes
		}
	}
}

edmp_interference_fromfrom_is_likely_to_interact_with_root = {
	ROOT_FROMFROM = {
		OR = {
			
			# Always include players (present or absent)
			edmp_this_character_is_a_current_or_absent_human_player = yes
			
			# Always include rulers who are raiding ROOT, or who ROOT is raiding, or who is at war with ROOT
			AND = {
				is_ruler = yes
				war_with = ROOT
			}
			
			# Normal case: Include anyone who could theoretically be declared on (if a relevant CB existed)
			AND = {
				
				# Only rulers in diplo range of ROOT
				is_ruler = yes
				is_within_diplo_range = ROOT
				
				# Non-merc or landed merc. No unlanded mercs!
				OR = {
					is_landed = yes
					mercenary = no
				}
				
				OR = {                                                             # One of the following is true:
					independent = yes                                          #  --- FROMFROM is independent (ROOT can always declare on indep FROMFROM; but can only declare on non-indep FROMFROM in special cases - see below)
					liege = { character = ROOT }                               #  --- ROOT is FROMFROM's direct liege
					any_vassal = { character = ROOT }                          #  --- ROOT is FROMFROM's direct vassal
					liege = { ROOT = { liege = { character = PREVPREV } } }    #  --- ROOT and FROMFROM have the same de facto liege
				}
				
			}
		}
	}
}

edmp_interference_this_is_not_some_kind_of_temporary_title = {
	custom_tooltip = {
		text = edmp_interference_this_is_not_some_kind_of_temporary_title_tt
		temporary              = no
		adventurer             = no
		mercenary              = no
		holy_order             = no
		rebel                  = no
		is_landless_type_title = no
	}
}

edmp_interference_win_defensive_wars_is_available_to_this_character = {
	
	war = yes
	
	OR = {
		# Anyone (human or AI) specifically designated by the host can use this feature
		has_character_modifier = edmp_interference_win_defensive_wars
		has_character_modifier = edmp_interference_win_defensive_wars_perm
		# Global rule affects current players, or absent players, or both
		AND = {
			custom_tooltip = {
				text = edmp_interference_win_defensive_wars_global_human_is_active_tt
				has_global_flag = edmp_interference_win_defensive_wars_global_human
			}
			edmp_this_character_is_a_current_human_player = yes
		}
		AND = {
			custom_tooltip = {
				text = edmp_interference_win_defensive_wars_global_absent_player_is_active_tt
				has_global_flag = edmp_interference_win_defensive_wars_global_absent_player
			}
			edmp_this_character_is_an_absent_human_player = yes
		}
	}
	
}

edmp_interference_win_defensive_wars_root_character_is_a_valid_target_for_from_to_win_war = {
	
	# ROOT must not be current/absent human (because only AI attackers can be forced to lose a war)
	ROOT = {
		show_scope_change = no
		edmp_this_character_is_not_a_current_or_absent_human_player = yes
	}
	
	# FROM has win defensive wars due to global rule; or FROM has been specially designated by an admin
	ROOT_FROM = {
		edmp_interference_win_defensive_wars_is_available_to_this_character = yes
	}
	
	# ROOT is **primary** attacker in a war where FROM is a **primary/secondary** defender
	custom_tooltip = {
		text = edmp_interference_win_defensive_wars_root_is_primary_attacker_and_from_is_a_defender_tt
		ROOT = {
			any_war = {
				attacker = { character = ROOT }
				any_defender = { character = ROOT_FROM }
			}
		}
	}
	
}
