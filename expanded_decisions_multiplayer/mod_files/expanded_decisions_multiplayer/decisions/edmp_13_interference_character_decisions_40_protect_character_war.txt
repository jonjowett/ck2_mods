		filter = all_including_me

			has_character_flag = edmp_interference_show

			is_ruler = yes
		}
		allow = { # Keep in sync with the allow clause below
			trigger_if = {
				limit = { has_global_flag = edmp_interference_win_defensive_wars_global_human }
				OR = {
					AND = {
						edmp_this_character_is_a_current_human_player = yes
						custom_tooltip = {
							text = edmp_interference_win_defensive_wars_global_human_is_not_active_tt
							always = no
						}
					}
					edmp_this_character_is_not_a_current_human_player = yes
				}
			}
			trigger_if = {
				limit = { has_global_flag = edmp_interference_win_defensive_wars_global_absent_player }

							text = edmp_interference_win_defensive_wars_global_absent_player_is_not_active_tt
							always = no
						}
					}
					edmp_this_character_is_not_an_absent_human_player = yes
				}
			}
			ROOT_FROM = { edmp_this_character_can_take_admin_helper_actions = yes }
		}
		effect = {
			hidden_tooltip = {
				ROOT_FROM     = { save_event_target_as = edmp_interference_decision_taker }
				ROOT          = { save_event_target_as = edmp_interference_singlechar_affected }
				set_global_flag = edmp_interference_singlechar_win_defensive_wars_grant
			}
			ROOT = {
				show_scope_change = no
				add_character_modifier = {
					name     = edmp_interference_win_defensive_wars
					years    = 3
					stacking = no
					# NB: We're not going to try super-hard to cover all possible future situations
					#     where modifier inheritance might break (eg. abdication),
					#     because this is only a temporary modifier, so the added complexity isn't really worthwhile.
					inherit  = yes
				}
				if = {
					limit = {
						ai = yes
						any_war = { any_defender = { character = ROOT } }
					}
					custom_tooltip = {
						text = edmp_interference_win_defensive_wars_ai_will_act_now_tt
						character_event = { id = edmp_wdw.2000 }
					}
				}
			}
			edmp_interference_notify_players_singlechar = yes
			hidden_tooltip = { clr_global_flag = edmp_interference_singlechar_win_defensive_wars_grant }
		}
	}
