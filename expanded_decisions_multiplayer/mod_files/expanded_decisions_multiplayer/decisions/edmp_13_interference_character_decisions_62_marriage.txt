	# Force marriage/betrothal between two eligible characters with the same home court.
	# If both characters are of age then they marry. Otherwise, they are betrothed.
	# Polarity of marriage can be changed afterwards, via one of the decisions above.
	# Polarity of betrothals CANNOT be set via scripting commands.
	edmp_interference_marry = {
		filter = all_including_me
		third_party_filter = home_court
		third_party = ROOT
		ai = no
		from_potential = {
			has_character_flag = edmp_interference_show
			edmp_this_character_can_see_admin_helper_actions = yes
		}
		potential = {
			ROOT = {
				# Keep this list in-sync with the allow clause below
				prisoner     = no
				is_married   = no
				is_betrothed = no
				OR = {
					is_consort = no
					is_senior_consort_party = yes
				}
			}
		}
		allow = {
			ROOT_FROM = { edmp_this_character_can_take_admin_helper_actions = yes }
			ROOT = {
				# Repeat ROOT's potential requirements, to help the user figure out the scope of the decision
				prisoner     = no
				is_married   = no
				is_betrothed = no
				OR = {
					is_consort = no
					is_senior_consort_party = yes
				}
				# Potentially-negative for both spouses
				edmp_this_character_is_valid_target_for_negative_actions_by_rootfrom = yes
			}
		}
		third_party_potential = {
			ROOT_FROMFROM = {
				# Keep this list in-sync with the third_party_allow clause below
				prisoner     = no
				is_betrothed = no
				is_married   = no
				OR = {
					is_consort = no
					is_senior_consort_party = yes
				}
				trigger_if   = { limit = { is_female = yes } ROOT = { is_female = no  } }
				trigger_else = {                             ROOT = { is_female = yes } }
			}
		}
		third_party_allow = {
			ROOT_FROMFROM = {
				# Repeat ROOT_FROMFROM's potential requirements, to help the user figure out the scope of the decision
				prisoner     = no
				is_betrothed = no
				is_married   = no
				OR = {
					is_consort = no
					is_senior_consort_party = yes
				}
				trigger_if   = { limit = { is_female = yes } ROOT = { is_female = no  } }
				trigger_else = {                             ROOT = { is_female = yes } }
				# Potentially-negative for both spouses
				edmp_this_character_is_valid_target_for_negative_actions_by_rootfrom = yes
			}
		}
		effect = {
			hidden_tooltip = {
				ROOT_FROM     = { save_event_target_as = edmp_interference_decision_taker    }
				ROOT          = { save_event_target_as = edmp_interference_bilateral_party_1 }
				ROOT_FROMFROM = { save_event_target_as = edmp_interference_bilateral_party_2 }
			}
			if = {
				limit = {
					ROOT          = { is_marriage_adult = yes }
					ROOT_FROMFROM = { is_marriage_adult = yes }
				}
				ROOT = {
					show_scope_change = no
					add_spouse = ROOT_FROMFROM
					hidden_tooltip = { set_global_flag = edmp_interference_bilateral_marry }
				}
			}
			else = {
				ROOT = {
					show_scope_change = no
					add_betrothal = ROOT_FROMFROM
					hidden_tooltip = { set_global_flag = edmp_interference_bilateral_betroth }
				}
			}
			edmp_interference_notify_players_bilateral = yes
			hidden_tooltip = {
				clr_global_flag = edmp_interference_bilateral_marry
				clr_global_flag = edmp_interference_bilateral_betroth
			}
		}
	}
