edmp_interference_notify_players_tooltip_only = {
	if = {
		limit = { edmp_this_is_a_multiplayer_game = yes }
		custom_tooltip = { text = edmp_interference_other_players_notified_tt }
	}
}

edmp_interference_notify_players_force_end_war_white_peace = {
	if = {
		limit = { edmp_this_is_a_multiplayer_game = yes }
		custom_tooltip = {
			text = edmp_interference_other_players_notified_tt
			event_target:edmp_interference_decision_taker = {
				any_player = {
					limit = {
						OR = {
							NOT = { character = PREV }
							has_character_flag = edmp_interference_debug_include_sender_in_notifications
						}
					}
					character_event = { id = edmp_ifn.1001 }
				}
			}
		}
	}
	hidden_tooltip = {
		clear_event_target = edmp_interference_force_end_war_white_peace_war
		clear_event_target = edmp_interference_force_end_war_white_peace_attacker
		clear_event_target = edmp_interference_force_end_war_white_peace_defender
		clear_event_target = edmp_interference_decision_taker
	}
}
