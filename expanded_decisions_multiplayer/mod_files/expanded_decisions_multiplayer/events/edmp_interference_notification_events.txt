# War forced to end by admin
character_event = {
	id    = edmp_ifn.1001
	title = edmp_ifn_title
	is_triggered_only = yes
	portrait = event_target:edmp_interference_decision_taker
	picture = GFX_evt_recieve_letter
	
	desc = {
		trigger = { has_global_flag = edmp_interference_force_end_war_white_peace }
		text    = edmp_ifn_1001_desc_white_peace
	}
	desc = {
		trigger = { has_global_flag = edmp_interference_force_end_war_invalid }
		text    = edmp_ifn_1001_desc_invalid
	}
	desc = {
		trigger = { has_global_flag = edmp_interference_force_end_war_defender_victory }
		text    = edmp_ifn_1001_desc_defender_victory
	}
	desc = {
		trigger = { has_global_flag = edmp_interference_force_end_war_attacker_victory }
		text    = edmp_ifn_1001_desc_attacker_victory
	}
	desc = {
		trigger = {
			NOR = {
				has_global_flag = edmp_interference_force_end_war_white_peace
				has_global_flag = edmp_interference_force_end_war_invalid
				has_global_flag = edmp_interference_force_end_war_defender_victory
				has_global_flag = edmp_interference_force_end_war_attacker_victory
			}
		}
		text    = edmp_ifn_1001_desc_fallback
	}
	
	# ROOT is primary or secondary attacker
	#  - Attacker victory = excellent
	#  - White peace or defender victory = curses (NB: white peace usually imposes some costs on the attacker)
	#  - Anything else = OK/affected
	# NB: ROOT's portrait can NEVER appear in the event option, so using show_portrait for both attacker and defender should never be confusing
	option = {
		trigger = {
			event_target:edmp_interference_force_end_war_war = { any_attacker = { character = ROOT } }
			has_global_flag = edmp_interference_force_end_war_attacker_victory
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		tooltip_info = yes
		name = edmp_ifn_evtopt_excellent
	}
	option = {
		trigger = {
			event_target:edmp_interference_force_end_war_war = { any_attacker = { character = ROOT } }
			OR = {
				has_global_flag = edmp_interference_force_end_war_defender_victory
				has_global_flag = edmp_interference_force_end_war_white_peace
			}
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		tooltip_info = yes
		name = edmp_ifn_evtopt_curses
	}
	option = {
		trigger = {
			event_target:edmp_interference_force_end_war_war = { any_attacker = { character = ROOT } }
			NOR = {
				has_global_flag = edmp_interference_force_end_war_attacker_victory
				has_global_flag = edmp_interference_force_end_war_defender_victory
				has_global_flag = edmp_interference_force_end_war_white_peace
			}
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		tooltip_info = yes
		name = edmp_ifn_evtopt_ok_affected
	}
	
	
	# ROOT is primary or secondary defender
	#  - Defender victory = excellent
	#  - Attacker victory = curses
	#  - Anything else = OK/affected
	# NB: ROOT's portrait can NEVER appear in the event option, so using show_portrait for both attacker and defender should never be confusing
	option = {
		trigger = {
			event_target:edmp_interference_force_end_war_war = { any_defender = { character = ROOT } }
			has_global_flag = edmp_interference_force_end_war_defender_victory
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		tooltip_info = yes
		name = edmp_ifn_evtopt_excellent
	}
	option = {
		trigger = {
			event_target:edmp_interference_force_end_war_war = { any_defender = { character = ROOT } }
			has_global_flag = edmp_interference_force_end_war_attacker_victory
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		tooltip_info = yes
		name = edmp_ifn_evtopt_curses
	}
	option = {
		trigger = {
			event_target:edmp_interference_force_end_war_war = { any_defender = { character = ROOT } }
			NOR = {
				has_global_flag = edmp_interference_force_end_war_defender_victory
				has_global_flag = edmp_interference_force_end_war_attacker_victory
			}
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		tooltip_info = yes
		name = edmp_ifn_evtopt_ok_affected
	}
	
	# ROOT is not involved in the war
	#  - Always OK/unaffected
	option = {
		trigger = {
			NOR = {
				event_target:edmp_interference_force_end_war_war = { any_attacker = { character = ROOT } }
				event_target:edmp_interference_force_end_war_war = { any_defender = { character = ROOT } }
			}
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		name = edmp_ifn_evtopt_ok_unaffected
	}
	
	# Fallback
	#  - In case there is no other visible option (eg. if one or more event_target scopes have not been set up correctly)
	option = {
		trigger = {
			# scope = { always = yes } returns true if and only if the scope exists
			# Therefore, NOT = { scope = { always = yes } } returns true if and only if the scope does NOT exist
			# So, this option will be visible if and only if any of the expected scopes have not been correctly defined
			OR = {
				NOT = { event_target:edmp_interference_force_end_war_attacker = { always = yes } }
				NOT = { event_target:edmp_interference_force_end_war_defender = { always = yes } }
				NOT = { event_target:edmp_interference_force_end_war_war      = { always = yes } }
			}
		}
		show_portrait = event_target:edmp_interference_force_end_war_attacker
		show_portrait = event_target:edmp_interference_force_end_war_defender
		name = edmp_ifn_evtopt_ok_unaffected_fallback
	}
	
	after = {
		hidden_tooltip = {
			clear_event_target = edmp_interference_decision_taker
			clear_event_target = edmp_interference_force_end_war_war
			clear_event_target = edmp_interference_force_end_war_attacker
			clear_event_target = edmp_interference_force_end_war_defender
			clr_global_flag    = edmp_interference_force_end_war_white_peace
			clr_global_flag    = edmp_interference_force_end_war_invalid
			clr_global_flag    = edmp_interference_force_end_war_defender_victory
			clr_global_flag    = edmp_interference_force_end_war_attacker_victory
		}
	}
}

		trigger = {
			event_target:edmp_interference_tributary_action_suzerain = { character = ROOT }
			NOT = { has_global_flag = edmp_interference_tributary_bad_for_suzerain }
		}
		show_portrait = event_target:edmp_interference_tributary_action_tributary
		tooltip_info = yes
		name = edmp_ifn_evtopt_excellent
	}
	
	# ROOT is tributary
	option = {
		trigger = {
			event_target:edmp_interference_tributary_action_tributary = { character = ROOT }
			has_global_flag = edmp_interference_tributary_bad_for_tributary
		}
		show_portrait = event_target:edmp_interference_tributary_action_suzerain
		tooltip_info = yes
		name = edmp_ifn_evtopt_curses
	}
	option = {
		trigger = {
			event_target:edmp_interference_tributary_action_tributary = { character = ROOT }
			NOT = { has_global_flag = edmp_interference_tributary_bad_for_tributary }
		}
		show_portrait = event_target:edmp_interference_tributary_action_suzerain
		tooltip_info = yes
		name = edmp_ifn_evtopt_excellent
	}
	
	# ROOT is neither tributary nor suzerain
	option = {
		trigger = {
			NOR = {
				event_target:edmp_interference_tributary_action_tributary = { character = ROOT }
				event_target:edmp_interference_tributary_action_suzerain = { character = ROOT }
			}
		}
		show_portrait = event_target:edmp_interference_tributary_action_tributary
		show_portrait = event_target:edmp_interference_tributary_action_suzerain
		name = edmp_ifn_evtopt_ok_unaffected
	}
	
	after = {
		hidden_tooltip = {
			
			clear_event_target = edmp_interference_decision_taker
			
			clear_event_target = edmp_interference_tributary_action_tributary
			clear_event_target = edmp_interference_tributary_action_suzerain
			
			clr_global_flag    = edmp_interference_tributary_bad_for_tributary
			clr_global_flag    = edmp_interference_tributary_bad_for_suzerain
			
			clr_global_flag    = edmp_interference_tributary_cancelled
			clr_global_flag    = edmp_interference_tributary_forced
			clr_global_flag    = edmp_interference_tributary_vassal_forced
			clr_global_flag    = edmp_interference_tributary_independence_forced
			
		}
	}
