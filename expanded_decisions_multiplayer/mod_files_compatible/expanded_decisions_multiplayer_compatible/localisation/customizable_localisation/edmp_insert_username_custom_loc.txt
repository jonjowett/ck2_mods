defined_text = {
	name = edmp_insert_username_from
	use_first_valid = yes
	
	# FROM is human, in a genuine MP game => Show FROM's username
	text = {
		trigger = {
			multiplayer = yes
			FROM = { ai = no }
		}
		localisation_key = edmp_insert_username_from_actual
	}
	
	# FROM is an absent player, in a genuine MP game => Show their "absent player" status
	text = {
		trigger = {
			multiplayer = yes
			FROM = { edmp_this_character_is_an_absent_human_player = yes }
		}
		localisation_key = edmp_insert_username_absent_player
	}
	
	# FROM is human/absent, in SP, but with MP debug mode turned on => Show a debug message (usernames only exist in MP)
	text = {
		trigger = {
			edmp_this_is_a_multiplayer_game = yes
			FROM = { edmp_this_character_is_a_current_or_absent_human_player = yes }
		}
		localisation_key = edmp_insert_username_sp_debug
	}
	
	# Any other situation => FROM is AI, or this is a SP game => Show nothing
	# (Since we actively want to show *nothing*, we can simply not write any code here!)
	
}
