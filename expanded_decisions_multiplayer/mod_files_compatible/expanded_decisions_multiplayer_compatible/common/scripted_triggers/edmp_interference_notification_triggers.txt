edmp_ifn_this_character_is_roots_enemy_for_notification_flavour = {
	
	OR = {
		
		# Enemy or rival (from opinion modifier properties)
		is_foe = ROOT
		
		# ROOT is plotting against this character
		is_plot_target_of = ROOT
		
		# ROOT and this character are directly at war with each other
		any_war = {
			attacker = { OR = { character = PREVPREV character = ROOT } }
			defender = { OR = { character = PREVPREV character = ROOT } }
		}
		
		# This character is doing something hostile to ROOT which is only captured by cchecking for specific opinion modifiers
		ROOT = {
			OR = {
				has_opinion_modifier = { who = PREV name = opinion_targeted_by_adventurer }    # This character is preparing an adventure against ROOT
				has_opinion_modifier = { who = PREV name = preparing_to_invade_me }            # This character is preparing an invasion against ROOT
				has_opinion_modifier = { who = PREV name = opinion_looted }                    # This character is raiding ROOT
			}
		}
		
	}
	
}

		dynasty = ROOT
		is_married = ROOT
		is_consort = ROOT

		is_allied_with = ROOT
	}
}
