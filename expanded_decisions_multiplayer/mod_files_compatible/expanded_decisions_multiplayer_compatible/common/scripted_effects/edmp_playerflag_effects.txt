edmp_playerflag_add_flag_to_this_character = {
	custom_tooltip = {
		text = edmp_playerflag_add_flag_to_this_character_tt
		add_character_modifier = {
			name     = edmp_playerflag
			duration = -1
			inherit  = yes
		}
		add_trait = edmp_absent_human_player # Ensures AI rationality, honor & lack of ambition
	}
}

edmp_playerflag_remove_flag_from_this_character = {
	custom_tooltip = {
		text = edmp_playerflag_remove_flag_from_this_character_tt
		remove_character_modifier = edmp_playerflag
		remove_trait = edmp_absent_human_player
	}
}

# NB: Pause/unpause admin helpers uses a modifier on province 1, instead of a global flag,
#        because playtesting showed that the global flag is often not propagated to other players. (???!!!)
edmp_playerflag_adminhelpers_pause_effect = {
	custom_tooltip = {
		text = edmp_playerflag_adminhelpers_pause_effect_tt
		1 = {
			add_province_modifier = {
				name = edmp_admin_helpers_are_paused
				duration = -1
				# Temporarily visible to help with debugging
				#hidden = yes
				stacking = no
			}
		}
	}
}
edmp_playerflag_adminhelpers_unpause_effect = {
	custom_tooltip = {
		text = edmp_playerflag_adminhelpers_unpause_effect_tt
		1 = { remove_province_modifier = edmp_admin_helpers_are_paused }
	}
}
