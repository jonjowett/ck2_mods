on_startup = {
	events = {
		EDMPPlayerFlag.1000 # Automatically flag all human players
		EDMPPlayerFlag.3000 # Check & if necessary re-apply admin & adminhelper flags
	}
}

# Inherit player/admin/adminhelper flag on abdication
on_new_holder = {
	events = {
		EDMPPlayerFlag.2000
	}
}
on_new_holder_inheritance = {
	events = {
		EDMPPlayerFlag.2000
	}
} 
on_new_holder_usurpation = {
	events = {
		EDMPPlayerFlag.2000
	}
}
