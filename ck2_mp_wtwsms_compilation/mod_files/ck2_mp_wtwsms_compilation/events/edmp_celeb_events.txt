namespace = edmp_celeb

# Player uses popper on another player
character_event = {
	id                = edmp_celeb.1300
	title             = edmp_celeb_event_title
	desc              = edmp_celeb_1300_desc
	border            = GFX_event_normal_frame_economy # Green
	picture           = GFX_evt_jugglers
	portrait          = event_target:edmp_celeb_decision_taker
	is_triggered_only = yes
	option            = {
		name = edmp_celeb_evtopt_party_starts
		tooltip = {
			add_character_modifier = {
				name  = edmp_celeb_popper_consumed
				years = 10
			}
		}
		custom_tooltip = {
			text = edmp_celeb_popper_refill_realm_levies_in_45_days_tt
			character_event = { id = edmp_celeb.1301 days = 45 }
		}
	}
	after             = { hidden_tooltip = { clear_event_target = edmp_celeb_decision_taker } }
}

}
