




# Passthroughs, allowing large amounts of TPAuto mod detection info to be added to loc text with a single bracket command

defined_text = {
	name = TPAuto_ModDetection
	use_first_valid = yes
	text = {
		trigger = { always = yes }
		localisation_key = TPAuto_ModDetection_Actual
	}
}
defined_text = {
	name = TPAuto_ModDetection_TradeRoutes
	use_first_valid = yes
	text = {
		trigger = { always = yes }
		localisation_key = TPAuto_ModDetection_TradeRoutes_Actual
	}
}
defined_text = {
	name = TPAuto_ModDetection_Buildings
	use_first_valid = yes
	text = {
		trigger = { always = yes }
		localisation_key = TPAuto_ModDetection_Buildings_Actual
	}
}





# Trade routes
#  --- NB: any_province is expensive, but these scripts are only ever calculated for user-visible tooltips, so it's fine.
