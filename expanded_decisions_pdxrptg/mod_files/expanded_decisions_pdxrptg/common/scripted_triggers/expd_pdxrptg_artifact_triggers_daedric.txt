# This allows characters to activate daedric artifacts without revealing their secret religion.
# (Because there's a way for a character to permanently activate them even if they don't secretly follow the relevant religion.)

	custom_tooltip = {
		text = expd_pdxrptg_this_character_is_favoured_by_azura_tt
		OR = {
			# Audax Validator "." Ignore_NEXT
			secret_religion = daedra_azura
			has_character_flag = expd_pdxrptg_favoured_by_azura
		}
	}

expd_pdxrptg_this_character_has_enough_offmap_currency_to_purchase_daedric_artifact_from_root = {
	trigger_if = {
		limit = {
			# Audax Validator "." Ignore_NEXT
			is_immortal = yes
		}
		has_offmap_currency = { offmap = ROOT value = 7500 }
	}
	trigger_else_if = {
		limit = {
			# Audax Validator "." Ignore_NEXT
			is_long_lived = yes
		}
		has_offmap_currency = { offmap = ROOT value = 5000 }
	}
	trigger_else = {
		has_offmap_currency = { offmap = ROOT value = 2500 }
	}
}

expd_pdxrptg_this_character_has_double_the_offmap_currency_needed_to_purchase_daedric_artifact_from_root = { # Used to slow down the AI
	trigger_if = {
		limit = {
			# Audax Validator "." Ignore_NEXT
			is_immortal = yes
		}
		has_offmap_currency = { offmap = ROOT value = 15000 }
	}
	trigger_else_if = {
		limit = {
			# Audax Validator "." Ignore_NEXT
			is_long_lived = yes
		}
		has_offmap_currency = { offmap = ROOT value = 10000 }
	}
	trigger_else = {
		has_offmap_currency = { offmap = ROOT value = 5000 }
	}
}

expd_pdxrptg_this_character_really_wants_a_daedric_artifact = {
	OR = {
		trait = ambitious
		trait = adventurer
		expd_pdxrptg_this_character_has_trait_dragonborn_descendant = yes
	}
}
