expd_pdxrptg_purchase_pet_initialise = {
	if = {
		limit = {
			OR = {
				NOT = { has_character_flag = expd_pdxrptg_purchase_pets_initialised }
				had_character_flag = { flag = expd_pdxrptg_purchase_pets_initialised months >= 1 } # Fallback in case something gets stuck...
			}
		}
		
		expd_pdxrptg_purchase_pets_clear_everything_except_decision_suppression = yes
		
		expd_pdxrptg_purchase_pets_setup_costs_for_this_character = yes
		
		expd_pdxrptg_purchase_pets_ai_choice_select_3 = yes
		
		if = {
			limit = {
				ai = no
				NOT = { has_global_flag = expd_pdxrptg_purchase_pets_debug_ai }
			}
			set_character_flag = expd_pdxrptg_purchase_pets_human
			set_variable = { which = pet_page_no value = 0 }
		}
		
		set_character_flag = expd_pdxrptg_purchase_pets_initialised
		
	}
}

expd_pdxrptg_purchase_pets_clear_everything = {
	hidden_tooltip = {
		expd_pdxrptg_purchase_pets_clear_everything_except_decision_suppression = yes
		clr_character_flag = buying_a_pet
	}

expd_pdxrpt_purchase_pet_round_final_cost = {
	# remainder(x) = x mod 1
	# if remainder(x) >= 0.5, we want to round up,   ie. round(x) = x + 1 - remainder(x)
	# else,                   we want to round down, ie. round(x) = x     - remainder(x)
	set_variable    = { which = pet_cost_final_to_round_remainder which = pet_cost_final_to_round }
	modulo_variable = { which = pet_cost_final_to_round_remainder value = 1 }
	if = {
		limit = { check_variable = { which = pet_cost_final_to_round_remainder value >= 0.5 } }
		change_variable = { which = pet_cost_final_to_round value = 1 }
	}
	subtract_variable = { which = pet_cost_final_to_round which = pet_cost_final_to_round_remainder }
}
