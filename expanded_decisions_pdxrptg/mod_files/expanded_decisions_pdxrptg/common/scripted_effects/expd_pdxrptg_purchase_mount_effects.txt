expd_pdxrptg_purchase_mounts_initialise = {
	if = {
		limit = {
			OR = {
				NOT = { has_character_flag = expd_pdxrptg_purchase_mounts_initialised }
				had_character_flag = { flag = expd_pdxrptg_purchase_mounts_initialised months >= 1 } # Fallback in case something gets stuck...
			}
		}
		
		expd_pdxrptg_purchase_mounts_clear_everything_except_decision_suppression = yes
		
		expd_pdxrptg_purchase_mounts_setup_costs_for_this_character = yes
		
		expd_pdxrptg_purchase_mounts_ai_choice_select_3 = yes
		
		if = {
			limit = {
				ai = no
				NOT = { has_global_flag = expd_pdxrptg_purchase_mounts_debug_ai }
			}
			set_character_flag = expd_pdxrptg_purchase_mounts_human
			set_variable = { which = mount_page_no value = 0 }
		}
		
		set_character_flag = expd_pdxrptg_purchase_mounts_initialised
		
	}
}

	# remainder(x) = x mod 1
	# if remainder(x) >= 0.5, we want to round up,   ie. round(x) = x + 1 - remainder(x)
	# else,                   we want to round down, ie. round(x) = x     - remainder(x)

	modulo_variable = { which = mount_cost_final_to_round_remainder value = 1 }
	if = {
		limit = { check_variable = { which = mount_cost_final_to_round_remainder value >= 0.5 } }
		change_variable = { which = mount_cost_final_to_round value = 1 }
	}
	subtract_variable = { which = mount_cost_final_to_round which = mount_cost_final_to_round_remainder }
}
