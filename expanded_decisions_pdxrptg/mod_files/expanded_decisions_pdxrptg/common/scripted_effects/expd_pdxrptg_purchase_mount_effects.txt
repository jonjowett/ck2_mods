expd_pdxrptg_purchase_mounts_initialise = {
	if = {
		limit = { NOT = { has_character_flag = expd_pdxrptg_purchase_mounts_initialised } }
		
		expd_pdxrptg_purchase_mounts_clear_everything_except_decision_suppression = yes
		
		expd_pdxrptg_purchase_mounts_setup_costs_for_this_character = yes
		
		expd_pdxrptg_purchase_mounts_ai_choice_select_3 = yes
		
		expd_pdxrptg_purchase_mounts_create_negative_costs = yes
		
		if = {
			limit = {
				ai = no
				NOT = { has_global_flag = expd_pdxrptg_purchase_mounts_debug_ai }
			}
			set_character_flag = expd_pdxrptg_purchase_mounts_human
			set_variable = { which = mount_page_no value = 0 }
		}
		
		set_character_flag = expd_pdxrptg_purchase_mounts_initialised
		
	}
}

expd_pdxrptg_purchase_mounts_setup_costs_for_this_character_continent = {
	
	# Increase cost significantly if mount is not from your continent
	
	
	
	
	
	# Tamriel
	
	if = {
		limit = {
			capital_scope = {
				# Audax Validator "." Ignore_NEXT
				region = nirn_tamriel
			}
		}
		# Capital is in Tamriel => No surcharge
		set_variable = { which = mount_tamriel_cost_mult value = 1 }
	}
	else_if = {
		limit = {
			any_demesne_province = {
				# Audax Validator "." Ignore_NEXT
				region = nirn_tamriel
			}
		}
		# Capital is NOT in Tamriel, but a demesne province is => small surcharge
		set_variable = { which = mount_tamriel_cost_mult value = 2 }
	}
	else_if = {
		limit = {
			any_realm_province = {
				# Audax Validator "." Ignore_NEXT
				region = nirn_tamriel
			}
		}
		# Capital is NOT in Tamriel, but a subrealm province is => medium surcharge
		set_variable = { which = mount_tamriel_cost_mult value = 3 }
	}
	else_if = {
		limit = {
			top_liege = {
				any_realm_province = {
					# Audax Validator "." Ignore_NEXT
					region = nirn_tamriel
				}
			}
		}
		# No subrealm province is in Tamriel, but your top liege's realm has one => large surcharge
		set_variable = { which = mount_tamriel_cost_mult value = 5 }
	}
	else_if = {
		limit = {
			top_liege = {
				any_realm_province = {
					OR = {
						# Audax Validator "." Ignore_NEXT
						region = nirn_yokuda
						# Audax Validator "." Ignore_NEXT
						region = nirn_atmora
						# Audax Validator "." Ignore_NEXT
						region = nirn_akavir_cathnoquey
						# Audax Validator "." Ignore_NEXT
						region = nirn_roscrea
					}
				}
			}
		}
		# Nearest top liege realm province is in an adjacent subcontinent to Tamriel => very large surcharge
		set_variable = { which = mount_tamriel_cost_mult value = 10 }
	}
	else = {
		# Nearest top liege realm province is NOT in an adjacent subcontinent to Tamriel => extremely large surcharge
		set_variable = { which = mount_tamriel_cost_mult value = 20 }
	}
	multiply_variable = { which = mount_rouncey_cost     which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_jennet_cost      which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_mountain_cost    which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_destrier_cost    which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_courser_cost     which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_palfrey_cost     which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_hobbeye_cost     which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_camel_h_cost     which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_guar_c_cost      which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_guar_f_cost      which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_guar_s_cost      which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_kagouti_cost     which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_shellback_cost   which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_pahmar_cost      which = mount_tamriel_cost_mult }
	multiply_variable = { which = mount_senche_cost      which = mount_tamriel_cost_mult }
	
	
	
	
	
	# Akavir
	
	if = {
		limit = {
			capital_scope = {
				# Audax Validator "." Ignore_NEXT
				region = nirn_akavir
			}
		}
		# Capital is in Akavir => No surcharge
		set_variable = { which = mount_akavir_cost_mult value = 1 }
	}
	else_if = {
		limit = {
			any_demesne_province = {
				# Audax Validator "." Ignore_NEXT
				region = nirn_akavir
			}
		}
		# Capital is NOT in Akavir, but a demesne province is => small surcharge
		set_variable = { which = mount_akavir_cost_mult value = 2 }
	}
	else_if = {
		limit = {
			any_realm_province = {
				# Audax Validator "." Ignore_NEXT
				region = nirn_akavir
			}
		}
		# Capital is NOT in Akavir, but a subrealm province is => medium surcharge
		set_variable = { which = mount_akavir_cost_mult value = 3 }
	}
	else_if = {
		limit = {
			top_liege = {
				any_realm_province = {
					# Audax Validator "." Ignore_NEXT
					region = nirn_akavir
				}
			}
		}
		# No subrealm province is in Akavir, but your top liege's realm has one => large surcharge
		set_variable = { which = mount_akavir_cost_mult value = 5 }
	}
	else_if = {
		limit = {
			top_liege = {
				any_realm_province = {
					OR = {
						# Audax Validator "." Ignore_NEXT
						region = nirn_atmora
						# Audax Validator "." Ignore_NEXT
						region = nirn_roscrea
					}
				}
			}
		}
		# Nearest top liege realm province is in an adjacent subcontinent to Akavir => very large surcharge
		set_variable = { which = mount_akavir_cost_mult value = 10 }
	}
	else = {
		# Nearest top liege realm province is NOT in an adjacent subcontinent to Akavir => extremely large surcharge
		set_variable = { which = mount_akavir_cost_mult value = 20 }
	}
	multiply_variable = { which = mount_cerigyn_cost     which = mount_akavir_cost_mult }
	multiply_variable = { which = mount_mijako_cost      which = mount_akavir_cost_mult }
	multiply_variable = { which = mount_camel_p_cost     which = mount_akavir_cost_mult }
	
}
