	#Azura
	oblivion_azuras_star = {
		only_playable = yes
		button_name = offmap_boons
		ai_check_interval = 1

		from_potential = {
			has_dlc = "Jade Dragon"
			mercenary = no
			holy_order = no
			is_landed = yes
			is_adult = yes
			worships_azura = yes
		}

		potential = {
			is_offmap_tag = oblivion_azura
		}
	
		allow = {
			show_only_failed_conditions = yes
			FROM = {
				show_scope_change = no
				OR = {
					trait = dragonborn
					AND = {
						worships_azura = yes
						has_secret_religion = no
					}
				}
				ROOT = { is_offmap_tag = oblivion_azura }
				
				trigger_if = {
					limit = { is_immortal = yes }
					has_offmap_currency = { offmap = ROOT value = 7500 }
				}
				
				trigger_else_if = {
					limit = { is_long_lived = yes }
					has_offmap_currency = { offmap = ROOT value = 5000 }
				}
				
				trigger_else = {
					has_offmap_currency = { offmap = ROOT value = 2500 }
				}
				prisoner = no
				NOT = { trait = incapable }
				is_inaccessible_trigger = no
			}
			AND = {
				ROOT = { is_offmap_tag = oblivion_azura }
				oblivion_azura_boon_decisions_enabled_trigger = yes
				oblivion_azura = {
					NOT = { has_status = oblivion_cataclysm }
					offmap_ruler = {
						has_artifact = azuras_star
					}
				}
			}
		}

		effect = {
			FROM = {
				show_scope_change = no
				sound_effect = china_grace_spend
				if = {
					limit = { is_immortal = yes }
					add_offmap_currency = {
						offmap = ROOT
						value = -7500
					}
				}
				
				else_if = {
					limit = { is_long_lived = yes }
					add_offmap_currency = {
						offmap = ROOT
						value = -5000
					}
				}
				
				else = {
					add_offmap_currency = {
						offmap = ROOT
						value = -2500
					}
				}
			}
			hidden_effect = {
				FROM = {
					#set_china_boon_cd_effect = yes
				}
				oblivion_azura = {
					offmap_ruler = {
						save_event_target_as = ek_artifact_origin
					}
				}
				event_target:ek_artifact_origin = {
					random_artifact = {
						limit = { artifact_type = azuras_star }
						transfer_artifact = {
							from = event_target:ek_artifact_origin
							to = FROM
						}
					}
				}
			}
		}

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0.1 # slow down
			}
			modifier = {
				FROM = {
					NOT = { has_grace_major_trigger = yes }
				}
				factor = 0.1
			}
		}
	}
