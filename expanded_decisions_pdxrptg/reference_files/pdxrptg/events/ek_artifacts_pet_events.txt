#Test event to make pets old
character_event = {
	id = ekpets.0
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		random_artifact = {
			limit = {
				OR = {
					has_artifact_flag = mount
					has_artifact_flag = pet
				}
				NOT = { has_artifact_flag = pet_death_imminent }
			}
			set_artifact_flag = pet_death_imminent
		}
	}
}

character_event = {
	id = ekpets.10
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_artifact = {
			has_artifact_flag = pet_death_imminent
		}
	}
	
	immediate = {
		random_artifact = {
			limit = { has_artifact_flag = pet_death_imminent }
			random_list = {
				30 = {
					modifier = {
						factor = 0
						has_artifact_flag = summoned_creature
					}
				} #Do nothing
				70 = {
					modifier = {
						factor = 0
						OR = {
							artifact_type = mount_senche
							artifact_type = mount_pahmar
							has_artifact_flag = summoned_creature
						}
					}
					save_event_target_as = ek_dead_pet
					owner = { character_event = { id = ekpets.15 } }
				}
				70 = {
					modifier = {
						factor = 0
						NOR = {
							artifact_type = mount_senche
							artifact_type = mount_pahmar
						}
					}
					save_event_target_as = ek_dead_pet
					owner = { character_event = { id = ekpets.20 } }
				}
				70 = {
					modifier = {
						factor = 0
						NOT = {
							has_artifact_flag = summoned_creature
						}
					}
					save_event_target_as = ek_dead_pet
					owner = { character_event = { id = ekpets.25 } }
				}
			}
		}
	}
}

#Pet dies - based on vanilla Cat dying event
character_event = {
	id = ekpets.15
	desc = EKPets15
	picture = GFX_evt_garden
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { #Send this event if the player has WoL to clear flags
				has_dlc = "Way of Life"
				event_target:ek_dead_pet = {
					artifact_type = pet_dog_masterless
				}
			}
			character_event = { id = WoL.5291 }
		}
	}
	
	option = {
		name = EKPets15OptA
		event_target:ek_dead_pet = {
			destroy_artifact = yes
		}
	}
	option = {
		name = EKPets15OptB
		trigger = { scaled_wealth = 0.6 }
		event_target:ek_dead_pet = {
			destroy_artifact = yes
		}
		scaled_wealth = -0.6
		if = {
			limit = { 
				NOR = { 
					trait = greedy
					trait = charitable
				}
			}
			random = {
				chance = 20
				add_trait = charitable
			}
		}
		if = {
			limit = { trait = greedy }
			random = {
				chance = 25
				remove_trait = greedy
			}
		}
	}
	option = {
		name = EKPets15OptC
		trigger = {
			scaled_wealth = 0.25
			OR = {
				trait = hunter
				is_batshit_crazy = yes
			}
		}
		event_target:ek_dead_pet = {
			destroy_artifact = yes
		}
		scaled_wealth = -0.25
		prestige = 30
		if = {
			limit = {
				has_nickname = no
			}
			random = {
				chance = 20
				give_nickname = nick_the_keeper
			}
		}
	}
	option = {
		name = EKPets15OptD
		trigger = {
			has_dlc = "Reapers"
			has_character_modifier = in_seclusion
			is_ruler = yes
			event_target:ek_dead_pet = {
				has_artifact_flag = mount
				NOR = { #Otherwise people will be eating Khajiits. Implications...*unpleasant.*
					artifact_type = mount_senche
					artifact_type = mount_pahmar
				}
			}
		}
		event_target:ek_dead_pet = {
			destroy_artifact = yes
		}
		change_variable = { which = remaining_food value = 2 }
	}
}
