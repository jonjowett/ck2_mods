character_event = { #Delay event, so that children get their infancy traits properly translated to adult traits first
	id = ekgreathouses.0
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		has_house_trait_including_houseless = no
		NOT = { has_global_flag = great_houses_banned }
		OR = {
			NOT = { top_liege = { primary_title = { title = k_almsivi } } }
			AND = {
				controls_religion = yes
				religion = almsivi
			}
		}
		OR = { 
			mother_even_if_dead = { has_house_trait = yes }
			father_even_if_dead = { has_house_trait = yes }
			liege = { is_great_house_grandmaster = yes }
			AND = {
				top_liege = { has_house_trait_including_houseless = yes }
				liege = { has_house_trait_including_houseless = yes }
				OR = {
					culture = chimer
					culture = dunmer
				}
			}
			AND = {
				k_almsivi = { holder_scope = { has_house_trait_including_houseless = yes } }
				is_tributary = { type = tribunal_temple }
			}
		}
	}
	
	immediate = {
		character_event = { id = ekgreathouses.1 days = 60 }
	}
}
