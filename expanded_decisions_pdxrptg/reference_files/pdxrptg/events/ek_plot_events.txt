		}
		modifier = {
			factor = 0.5
			num_of_plot_backers = 10
		}

			factor = 0.5
			num_of_plot_backers = 20
		}
		modifier = {
			factor = 0.5

		}
	}
	immediate = {
		ROOT = { 
			save_event_target_as = target_plotter 
			activate_plot = yes
		}
		plot_target_char = { save_event_target_as = plot_target }
		any_plot_backer = { character_event = { id = EKplots.12 } }
	}
	option = { name = OK }
}
