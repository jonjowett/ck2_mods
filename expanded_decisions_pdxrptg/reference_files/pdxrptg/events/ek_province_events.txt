province_event = {
	id = kprov.1
	desc = kprov1d
	picture = GFX_evt_goblins
	
	immediate = {
		# Check which generic guild can be called
		if = {
			limit = {
				OR = {
					region = nirn_tamriel
					region = nirn_yokuda
				}
			}
			fighters_guild_tamriel = { save_event_target_as = generic_guild }
		}
		
		else_if = {
			limit = {
				region = nirn_akavir
			}
			fighters_guild_akavir = { save_event_target_as = generic_guild }
		}
		
		# Check which cultural guild can be called
		if = {
			limit = {
				region = nirn_tamriel_skyrim
			}
			companions_tamriel = { save_event_target_as = cultural_guild }
		}
		
		else_if = {
			limit = {
				OR = {
					region = nirn_tamriel_hammerfell
					region = nirn_yokuda
				}
			}
			ansei_guild_tamriel = { save_event_target_as = cultural_guild }
		}
		
		# Assuming the character can deal with the infestation themselves, should the AI do it?
		if = {
			limit = { owner = { combat_rating = 60 } }
			owner = { set_character_flag = should_hire_self }
		}
	}
	
	is_triggered_only = yes
