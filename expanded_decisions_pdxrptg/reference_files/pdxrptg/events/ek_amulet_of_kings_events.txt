#The "Currently Seeking the Amulet" Fork - Does the Amulet have an owner? or not?
character_event = {
	id = amuletofkings.10
	desc = {
		text = AmuletOfKings10A #Someone owns the Amulet
		picture = GFX_evt_whispers
		trigger = {
			has_global_flag = amulet_of_kings_active
		}
	}
	desc = {
		text = AmuletOfKings10B #No one has the Amulet
		picture = GFX_evt_into_the_dungeon
		trigger = {
			NOT = { has_global_flag = amulet_of_kings_active }
		}
	}
	picture = GFX_evt_into_the_dungeon
	
	is_triggered_only = yes
	
	immediate = { #Does anyone actually own the Chim-el Adabal?
		if = { #If the Amulet already is unearthed, ping its holder
			limit = { has_global_flag = amulet_of_kings_active }
			event_target:the_amulet_of_kings = {
				artifact_owner = {
					save_event_target_as = ek_amulet_owner
					set_character_flag = wants_coronation
				}
			}
		}
	}
	
	option = {
		name = AmuletOfKings10OptA
		trigger = { has_global_flag = amulet_of_kings_active }
		event_target:ek_amulet_owner = {
			character_event = { id = amuletofkings.15 }
		}
	}
	option = {
		name = AmuletOfKings10OptB
		trigger = { NOT = { has_global_flag = amulet_of_kings_active } }
		narrative_event = { id = amuletofkings.150 }
	}
}
character_event = {
	id = amuletofkings.15
	desc = AmuletOfKings15
	picture = GFX_evt_whispers
	
	is_triggered_only = yes
	
	option = { #Give the Amulet away
		name = AmuletOfKings15Accept
		ai_chance = {
			factor = 5
			
			modifier = { #Dragonblooded characters will never give the Amulet away.
				factor = 0
				OR = {
					is_former_emperor_bloodline = yes
					is_dragonblood = yes
				}
			}
			modifier = {
				factor = 0.1
				is_rival = FROM
			}
			modifier = {
				factor = 50
				is_friend = FROM
			}
		}
		random_artifact = {
			limit = { has_artifact_flag = amulet_of_kings_flag }
			transfer_artifact = {
				from = ROOT
				to = FROM
			}
		}
		hidden_tooltip = {
			clr_character_flag = wants_coronation
		}
		FROM = {
			character_event = { id = amuletofkings.20 } 
		}
	}
	option = { #Keep the Amulet - May be hazardous to your health
		name = AmuletOfKings15Decline
		ai_chance = {
			factor = 5
			
			modifier = {
				factor = 5
				trait = greedy
			}
			modifier = {
				factor = 5
				is_rival = FROM
			}
		}
		FROM = { character_event = { id = amuletofkings.25 } }
	}
	option = { #Keep the Amulet, but perform a sham Dragonfire coronation to falsely take credit
		name = AmuletOfKings15Compromise
		trigger = {
			independent = yes
			OR = {
				has_landed_title = c_imperialisle
				any_vassal = {
					has_landed_title = c_imperialisle
				}
			}
		}
		ai_chance = {
			factor = 5
			
			modifier = { #Dragonblooded characters will never give the Amulet away.
				factor = 0
				OR = {
					is_former_emperor_bloodline = yes
					is_dragonblood = yes
				}
			}
			modifier = {
				factor = 10
				trait = envious
			}
			modifier = {
				factor = 10
				trait = craven
			}
		}
		custom_tooltip = { text = AmuletOfKings15CompromiseTooltip }
		FROM = { character_event = { id = amuletofkings.30 } }
	}
}
