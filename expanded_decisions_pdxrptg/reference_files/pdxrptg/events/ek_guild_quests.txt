#Society/Guild quest events

namespace = ekguildquests

society_quest_event = {
	id = ekguildquests.10
	title = quest_fighters_goblins
	desc = EKGuildQuests10
	
	is_triggered_only = yes
	
	option = {
		name = ACCEPT
		ai_chance = {
			factor = 1
		}
		set_quest = quest_fighters_goblins
		character_event = { id = ekguildquests.11 days = 3 }
		
		roll_easy_pcs = yes
	}
	option = {
		name = DECLINE
		custom_tooltip = { text = EKGuildQuestsDeclineTooltip }
		ai_chance = {
			factor = 0
		}
		add_character_modifier = {
			name = quest_cooldown_timer
			hidden = yes
			days = 1000
		}
		hidden_tooltip = {
			FROM = {
				province_event = { id = ekguildquests.15 days = 60 random = 30 }
			}
		}
	}
}

character_event = {
	id = ekguildquests.11
	desc = EKGuildQuests11
	picture = GFX_evt_goblins
	
	is_triggered_only = yes
	
	option = {
		name = EKGuildQuests11Accept
		
		send_quest_result_pcs = yes
		
		if = {
			limit = { has_character_flag = guild_quest_fail }
			death = { death_reason = death_battle }
			hidden_tooltip = {
				FROMFROM = {
					province_event = { id = ekguildquests.14 days = 150 random = 30 }
				}
			}
		}
		
		else_if = {
			limit = { has_character_flag = guild_quest_maimed }
			add_maimed_trait_effect = yes
			character_event = { id = ekguildquests.12 }
		}
		
		else_if = {
			limit = { has_character_flag = guild_quest_wounded }
			add_trait = wounded
			character_event = { id = ekguildquests.12 }
		}
		
		
		else_if = {
			limit = { has_character_flag = guild_quest_success }
			character_event = { id = ekguildquests.12 }
		}
	}
}
