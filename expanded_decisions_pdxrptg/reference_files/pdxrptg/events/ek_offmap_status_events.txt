		modifier = { #Convalescing takes some time

				has_status = oblivion_convalescing
			}
		}
		modifier = { #Cataclysms are short-lived...usually
			factor = 10
			FROM = {
				has_status = oblivion_cataclysm
			}
		}
	}

		FROM = {
			if = {
				limit = {
					has_status = oblivion_normal
				}
				set_status = oblivion_apogee
			}
			if = {
				limit = {
					has_status = oblivion_convalescing
				}
				set_status = oblivion_normal
			}

				limit = {
					has_status = oblivion_cataclysm
