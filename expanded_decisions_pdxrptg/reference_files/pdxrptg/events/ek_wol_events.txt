		name = EKWOL10Decline

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Way of Life"
		has_focus = focus_scholarship
		prisoner = no
		NOT = { trait = incapable }
	}
	
	immediate = {
		random_list = {
			50 = { #50/50 chance of female/male
				create_character = {
					religion = ROOT
					culture = ROOT
					dynasty = none
					age = 55
					female = yes
					random_traits = yes
					attributes = {
						learning = 5
					}
					trait = mage_2
				}
				new_character = {
					#assign_birthsign = yes
					#assign_racial_trait = yes
					save_event_target_as = hired_daedric_scholar
					set_character_flag = no_court_invites
				}
			}
			50 = { #50/50 chance of female/male
				create_character = {
					religion = ROOT
					culture = ROOT
					dynasty = none
					age = 55

					random_traits = yes
					attributes = {
						learning = 5
					}
					trait = mage_2
				}

					#assign_birthsign = yes
					#assign_racial_trait = yes

					set_character_flag = no_court_invites

			}
		}

			limit = { has_character_flag = sent_for_azura_scholar }

				set_secret_religion = daedra_azura
