#Special event for ruler designer werewolves
character_event = {
	id = lycan.2
	hide_window = yes
	
	only_playable = yes
	min_age = 16
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { trait = lycan }
		OR = {
			trait = known_lycan
			trait = known_lycan_benevolent
		}
	}
	
	immediate = {
		add_trait = lycan
		if = {
			limit = {
				trait = lycan #If you're a lycanthrope, this is crucial
				trait = known_lycan #Known lycanthropes start at level 1 default
				NOR = { #if you don't have a lycan level, voila!
					has_character_flag = lycan_level_1
					has_character_flag = lycan_level_2
					has_character_flag = lycan_level_3
					has_character_flag = lycan_level_4
					has_character_flag = lycan_level_5
					has_character_flag = lycan_level_6
					has_character_flag = lycan_level_7
					has_character_flag = lycan_level_8
					has_character_flag = lycan_level_9
					has_character_flag = lycan_level_10
				}
			}
			set_character_flag = lycan_level_1
		}
		if = {
			limit = {
				trait = lycan #If you're a lycanthrope, this is crucial
				trait = known_lycan_benevolent #Benevolent lycanthropes start at level 5 default
				NOR = { #if you don't have a lycan level, voila!
					has_character_flag = lycan_level_1
					has_character_flag = lycan_level_2
					has_character_flag = lycan_level_3
					has_character_flag = lycan_level_4
					has_character_flag = lycan_level_5
					has_character_flag = lycan_level_6
					has_character_flag = lycan_level_7
					has_character_flag = lycan_level_8
					has_character_flag = lycan_level_9
					has_character_flag = lycan_level_10
				}
			}
			set_character_flag = lycan_level_5
		}
	}
}

character_event = {
	id = lycan.4
	desc = Lycan4
	picture = GFX_evt_burning_house
	
	capable_only = yes
	prisoner = no
	min_age = 16
	
	trigger = {
		location = { has_province_modifier = lycan_infestation }
		is_disease_immune = no
		NOT = { is_inaccessible_trigger = yes }
		NOT = { combat_rating = 40 }
	}

		months = 360
		
		modifier = {
			factor = 0.8
			trait = brave
		}
		modifier = {
			factor = 0.8
			trait = reckless
		}
		modifier = {
			factor = 1.5
			trait = craven
		}
		modifier = {
			factor = 1.5
			trait = paranoid
		}
	}
	
	option = {
		name = Lycan4Acknowledge
		hidden_tooltip = {
			random_list = {
				20 = { character_event = { id = lycan.5 } } #Character dies to lycans
				30 = { character_event = { id = lycan.6 } } #Character escapes unscathed
				50 = { character_event = { id = lycan.7 } } #Character gets infected
			}
		}	
	}
}

#Character survives but gets wounded, possibly infected
character_event = {
	id = lycan.7
	desc = Lycan7
	picture = GFX_evt_burning_house
	
	is_triggered_only = yes
	
	immediate = {
		roll_for_lycanthropy = yes
		if = { #Notify the liege if one exists
			limit = {
				independent = no
			}
		liege = { character_event = { id = lycan.10 } }
		}
	}
	
	option = {
		name = Lycan7Infection #Wounded, infected
		trigger = {
			is_disease_immune = no
			NOT = { has_character_flag = lycanthropy_immune }
			has_character_flag = lycan_disease
		}
		add_trait = wounded
		#hidden_tooltip = {
		#	long_character_event = { id = lycan.15 days = 3 }
		#}
	}
	option = {
		name = Lycan7NoInfection #Wounded, not infected
		trigger = {
			is_disease_immune = no
			NOT = { has_character_flag = lycanthropy_immune }
			NOT = { has_character_flag = lycan_disease }
		}
		add_trait = wounded
	}
	option = {
		name = Lycan7Immune #Wounded, but not infected due to immunity
		trigger = {
			OR = {
				is_disease_immune = yes
				has_character_flag = lycanthropy_immune
			}
		}
		add_trait = wounded
	}
}

	}

		}
	}
