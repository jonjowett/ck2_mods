### Traits related to religious behaviour
# These traits replace some direct religious references, and can allow finer opinion tweaks by changing who's defined as pro_necromancy, for example
# Each of these behaviours should be explicitly written in the religion features, in the localization files

pro_necromancy_trait = {
	random = no
	customizer = no
	
	is_visible = { has_global_flag = show_all_traits }
}

anti_necromancy_trait = {
	random = no
	customizer = no
	
	is_visible = { has_global_flag = show_all_traits }
}

pro_lycanthropy_trait = {
	random = no
	customizer = no
	
	is_visible = { has_global_flag = show_all_traits }
}
