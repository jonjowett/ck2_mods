improve_veterancy = {
	custom_tooltip = {
		text = improve_veterancy_tt
		hidden_tooltip = {
			trigger_switch = {
				on_trigger = trait
				veteran_leader_1 = { remove_trait = veteran_leader_1 add_trait = veteran_leader_2 }
				veteran_leader_2 = { remove_trait = veteran_leader_2 add_trait = veteran_leader_3 }
				veteran_leader_3 = { remove_trait = veteran_leader_3 add_trait = veteran_leader_4 }
				veteran_leader_4 = { remove_trait = veteran_leader_4 add_trait = veteran_leader_5 }
			}
			if = {
				limit = {
					NOR = {
						trait = veteran_leader_1
						trait = veteran_leader_2
						trait = veteran_leader_3
						trait = veteran_leader_4
						trait = veteran_leader_5
					}
				}
				add_trait = veteran_leader_1
			}
		}
	}
}
