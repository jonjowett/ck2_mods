# The limiting slots for equipped artifacts. Dynamic system (you can add new types of slots freely)
slots = {
	#Equipment
	head = 1	#Includes Crowns and Helmets
	neck = 1	#Amulets, Necklaces, Gorgets etc.
	cape = 1	#Capes, Mantlets
	clothing = 1 	#Clothing!
	torso = 1	#Armour, Robes.
	wrist = 2	#Can wear 2 bracelets, bracers
	ring = 2 	#Can wear up to 2 rings. Used to be 6, bringing it down for balancing purposes since rings are very strong and abundant
	gloves = 1	#Gloves & Gauntlets
	weapon = 1
	shield = 1	#Shields, Bucklers
	ranged_weapon = 1 #Bows, Crossbows, other ranged things
	boots = 1	#Boots & Greaves
	mask = 1 

	#Mounts/Pets&Familiars/"Slaves"
	pet = 1
	mount = 1
	retainer = 3

	#"special", Special Items, Books, Scrolls & Potions
	library = 3 #Can "wear" up to 3 books
	special = 3	#Can "wear" up to 3 "Special" items, these include lockboxs to protect against theft, certain miscellaneous items such as Orgnum's Coffer and other miscellaneous items.
	consumable = 3 #Can equip up to 3 consumables. Consumables are potions, scrolls and items that have no immediate effects but add a decision or will provide a new option in certain events. Taking those decisions or options will destroy the consumable and apply the effects of said consumable.
	
	
	#Not used in EK
	ceremonial_weapon = 1
	scepter = 1
	ceremonial_torso = 1
	
	#Unlike vanilla basically every artifact in EK must be equipped to have it's effects applied. Simply owning it won't provide benefits.
}
