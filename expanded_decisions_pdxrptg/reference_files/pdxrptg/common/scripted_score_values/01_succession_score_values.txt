# Consolidated identical and near-identical vote weights into scripted scores

# Root scope - Voter, the character voting in the election
# From scope - Candidate, Character considered for the title
# FromFrom scope - The title the nomination is for
# FromFromFrom scope - The title owner's candidate

# Elector voting themself
succession_vote_self_score = {
	additive_modifier = {
		value = 15
		localisation_key = tooltip_feudal_elector_vote_myself
		succession_vote_myself_trigger = yes
	}
	additive_modifier = {
		value = -25
		localisation_key = tooltip_feudal_elector_vote_not_myself
		FROMFROM = {
			NOT = {
				# Tanistry has a different weight
				has_law = succ_tanistry
			}
		}
		succession_vote_not_myself_trigger = yes
	}
	additive_modifier = {
		value = 40
		localisation_key = tooltip_feudal_elector_vote_myself_heir
		succession_vote_myself_heir = yes
	}
	additive_modifier = {
		value = 40
		localisation_key = tooltip_feudal_elector_vote_myself_ambitious
		succession_vote_myself_ambitious = yes
		is_adult = yes
	}
}

#Current ruler will always try to elect his player heir.
succession_player_heir_score = {
	additive_modifier = {
		value = 50
		localisation_key = tooltip_feudal_elector_vote_player_heir
		player_heir = {
			character = FROM
			NOT = { character = ROOT }
		}
		FROMFROM = {
			holder_scope = {
				character = PREVPREV
			}
			NOT = {
				is_rival = FROM #Unless is rival.
			}
		}
	}
}
