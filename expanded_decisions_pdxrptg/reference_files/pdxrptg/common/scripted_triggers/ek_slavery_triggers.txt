province_has_slaves = {
	has_province_modifier = province_has_slaves
}

province_is_enslaved = {
	has_province_modifier = province_enslaved
}

culture_can_be_enslaved = {
	NOR = {
		culture_group = undead
		culture_group = daedra
		culture_group = sload_group
	}
}

province_can_be_enslaved = {
	culture_can_be_enslaved = yes
	NOT = { province_is_enslaved = yes }
}

character_is_known_liberator = {
	OR = {
		trait = liberator
		any_owned_bloodline = {
			has_bloodline_flag = liberator_bloodline
			bloodline_is_active_for = PREV
		}
	}
}

character_is_known_slaver = {
	OR = {
		trait = slaver
		any_owned_bloodline = {
			has_bloodline_flag = slaver_bloodline
			bloodline_is_active_for = PREV
		}
	}
}

character_dabbles_in_slavery = {
	OR = {
		has_law = slavery_1
		check_variable = { which = numeroussecretslaveraidsonargonia value = 1 }
	}
}
