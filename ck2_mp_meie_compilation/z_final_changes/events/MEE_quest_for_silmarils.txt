### Quest for missing silmaril

	id = silmarils.1
	desc = "EVTDESCsilmarils.1"
	picture = GFX_evt_council

	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAsilmarils.1" ## My Marshall!
		trigger = {
			any_courtier_or_vassal = {
				has_job_title = job_marshal
				prisoner = no
			}
		}
		any_courtier_or_vassal = {
			limit = {
				has_job_title = job_marshal
			}
			character_event = { id =  silmarils.2 }
		}

	option = {
		name = "EVTOPTBsilmarils.1" ## My Spymaster!
		trigger = {
			any_courtier_or_vassal = {
				has_job_title = job_spymaster

			}
		}

			limit = {
				has_job_title = job_spymaster
			}
			character_event = { id =  silmarils.2 }
		}
		wealth = -400		
	}
	option = {
		name = "EVTOPTDsilmarils.1" ## Forgot about it...
		clr_character_flag = quest_for_missing_silmaril
