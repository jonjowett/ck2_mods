		effect = {

			set_character_flag = improve_skills_craft_timeout

			# Reset the timer, for the purposes of had_character_flag above
			clr_character_flag = improve_skills_craft
			set_character_flag = improve_skills_craft
			
			# Clear tracking flags for measuring success of this attempt
			clr_character_flag = helped_by_master
			clr_character_flag = not_helped_by_master

			clr_character_flag = bought_ancient_weapons
			clr_character_flag = bought_runic_objects
			clr_character_flag = bought_materials_forge
			
			# Trigger the event, and deduct the society currency
			hidden_effect = { character_event = { id = gwaith.0 days = 5 } }
			change_society_currency = {
				society = gwaith_i_mirdain
				value = -300
			}
			
		}

		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 0.1
			modifier = {
				is_ruler = yes
				factor = 10
			}
			modifier = {
				stewardship = 12
    			factor = 10
			}
			modifier = {
				stewardship = 15
    			factor = 10
			}

		}
	}
