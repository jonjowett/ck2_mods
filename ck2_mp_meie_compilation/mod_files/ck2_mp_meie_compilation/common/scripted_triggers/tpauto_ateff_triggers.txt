tpauto_ateff_mod_is_active = {
	
	# Check a couple of global flags which seem like they should only exist in ATEFF, and which should persist for the whole game.
	has_global_flag = scenario_initialized # From misc.5  in ...\events\misc_AfterTheEnd.txt ; always created on new game start
	has_global_flag = bad_checksum         # From misc.10 in ...\events\misc_AfterTheEnd.txt ; bad checksum is GUARANTEED if TPAuto is running at the same time!
	
	# Unfortunately, the names of those flags seem a bit generic.
	# So, in addition, test for a couple of game rules that seem unique to ATEFF.
	calc_true_if = {
		amount >= 4 # 4/5 is enough to be pretty certain, while allowing redundancy in case of typos or future ATEFF mod changes
		OR = {
			has_game_rule = { name = playercol value = historical }
			has_game_rule = { name = playercol value = customized }
		}
		OR = {
			has_game_rule = { name = redcoat_invasion value = historical     }
			has_game_rule = { name = redcoat_invasion value = random         }
			has_game_rule = { name = redcoat_invasion value = delayed_random }
			has_game_rule = { name = redcoat_invasion value = off            }
		}
		OR = {
			has_game_rule = { name = renovator_invasion value = historical     }
			has_game_rule = { name = renovator_invasion value = random         }
			has_game_rule = { name = renovator_invasion value = off            }
		}
		OR = {
			has_game_rule = { name = genghis_cree_invasion value = historical     }
			has_game_rule = { name = genghis_cree_invasion value = random         }
			has_game_rule = { name = genghis_cree_invasion value = off            }
		}
		OR = {
			has_game_rule = { name = colonisation_invasions value = historical     }
			has_game_rule = { name = colonisation_invasions value = random         }
			has_game_rule = { name = colonisation_invasions value = off            }
		}
	}
	
}
