decisions = {
	
	
	
	csh_update_all_characters = {
		ai = no
		potential = {
			NOT = { has_global_flag = csh_updates_disabled }
			csh_main_decisions_are_visible_for_this_character = yes
			csh_this_character_can_take_admin_actions = yes
		}
		allow = { always = yes }
		effect = {
			csh_update_all_traits_for_all_characters = yes
		}
		ai_will_do = { factor = 0 }
	}
	
	
	
	csh_remove_all_traits_and_disable_updates = {
		ai = no
		potential = {
			NOT = { has_global_flag = csh_updates_disabled }
			csh_main_decisions_are_visible_for_this_character = yes
			csh_this_character_can_take_admin_actions = yes
		}
		allow = { always = yes }
		effect = {
			custom_tooltip = {
				text = csh_disable_updates_tt
				set_global_flag = csh_updates_disabled
			}
			custom_tooltip = { text = csh_big_separator }
			custom_tooltip = {
				text = csh_remove_all_traits_tt
				any_character = { csh_remove_all_traits_from_this_character = yes }
			}
		}
		ai_will_do = { factor = 0 }
	}
	
	
	
	csh_reenable_updates = {
		ai = no
		potential = {
			has_global_flag = csh_updates_disabled
			csh_main_decisions_are_visible_for_this_character = yes
			csh_this_character_can_take_admin_actions = yes
		}
		allow = { always = yes }
		effect = {
			custom_tooltip = {
				text = csh_reenable_updates_tt
				clr_global_flag = csh_updates_disabled
			}
			custom_tooltip = { text = csh_big_separator }
			csh_update_all_traits_for_all_characters = yes
		}
		ai_will_do = { factor = 0 }
	}
	
	
	
	csh_disable_traits_for_me_only = {
		ai = no
		potential = {
			csh_main_decisions_are_visible_for_this_character = yes
			NOT = { has_character_modifier = csh_disable_traits_for_me_only_modifier }
			# The decision to hide the traits is only visible in multiplayer
			multiplayer = yes
		}
		allow = { always = yes }
		effect = {
			custom_tooltip = {
				text = csh_disable_traits_for_me_only_tt
				add_character_modifier = {
					name     = csh_disable_traits_for_me_only_modifier
					duration = -1
					hidden   = yes
					stacking = no
					inherit  = yes
				}
			}
		}
		ai_will_do = { factor = 0 }
	}
	csh_enable_traits_for_me_only = {
		ai = no
		potential = {
			csh_main_decisions_are_visible_for_this_character = yes
			has_character_modifier = csh_disable_traits_for_me_only_modifier
			# The decision to show the traits is visible in both MP and SP (so there's a way for players in SP to reverse a decision made in MP)
		}
		allow = { always = yes }
		effect = {
			custom_tooltip = {
				text = csh_enable_traits_for_me_only_tt
				remove_character_modifier = csh_disable_traits_for_me_only_modifier
			}
		}
		ai_will_do = { factor = 0 }
	}
	
	
	
}
