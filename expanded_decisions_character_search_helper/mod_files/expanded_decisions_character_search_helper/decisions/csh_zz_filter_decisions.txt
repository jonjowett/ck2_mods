decisions = {
	
	
	
	# Since there are a lot of filters, here are some "master" decisions which show/hide all filters
	
	csh_artifact_filter_show_filters = {
		ai = no
		potential = {
			csh_main_decisions_are_visible_for_this_character = yes
			NOT = { has_global_flag = csh_updates_disabled }
			NOT = { has_character_flag = csh_show_artifact_filters }
		}
		allow = { always = yes }
		effect = {
			hidden_tooltip = { set_character_flag = csh_show_artifact_filters }
		}
	}
	
	csh_artifact_filter_hide_filters = {
		ai = no
		potential = {
			csh_artifact_filters_are_visible_for_this_character = yes
		}
		allow = { always = yes }
		effect = {
			hidden_tooltip = { clr_character_flag = csh_show_artifact_filters }
		}
	}
	
	
	
	
	
	# Another "master" decision: Toggle all filters on/off
	
	csh_artifact_filter_toggle_all_on = {
		ai = no
		potential = {
			csh_artifact_filters_are_visible_for_this_character = yes
			csh_this_character_has_all_artifact_filters_disabled = yes
		}
		allow = { always = yes }
		effect = {
			csh_add_all_artifact_filters_to_this_character = yes
			custom_tooltip = { text = csh_artifact_filter_blank_line_tt }
			custom_tooltip = { text = csh_artifact_filter_notes_tt }
		}
	}
	
	csh_artifact_filter_toggle_all_off = {
		ai = no
		potential = {
			csh_artifact_filters_are_visible_for_this_character = yes
			csh_this_character_has_at_least_one_artifact_filter_enabled = yes
		}
		allow = { always = yes }
		effect = {
			csh_remove_all_artifact_filters_from_this_character = yes
			custom_tooltip = { text = csh_artifact_filter_blank_line_tt }
			custom_tooltip = { text = csh_artifact_filter_notes_tt }
		}
	}
	
	csh_artifact_filter_invert_all = {
		ai = no
		potential = {
			csh_artifact_filters_are_visible_for_this_character = yes
		}
		allow = {
			csh_this_character_has_at_least_one_artifact_filter_enabled = yes
			csh_this_character_does_not_have_all_artifact_filters_enabled = yes
		}
		effect = {
			csh_invert_this_characters_artifact_filters = yes
			custom_tooltip = { text = csh_artifact_filter_blank_line_tt }
			custom_tooltip = { text = csh_artifact_filter_notes_tt }
		}
	}
