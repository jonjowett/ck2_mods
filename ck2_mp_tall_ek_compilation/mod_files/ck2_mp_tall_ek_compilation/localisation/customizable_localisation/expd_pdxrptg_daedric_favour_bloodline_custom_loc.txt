	name = GetBloodlineFavour_Azura
	use_first_valid = yes
	text = {
		trigger = {
			ROOT = {
				any_owned_bloodline = {
					has_bloodline_flag = expd_pdxrptg_oblivion_touched

				}
			}
		}
		# Audax Validator "." Ignore_NEXT
		localisation_key = azura_azura
	}
}
