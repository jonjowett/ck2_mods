		limit = {
			# Audax Validator "." Ignore_NEXT

			# ROOT is not recognised as a valid offmap_power by the Validator v1.38.3 - MANY false positives

			value = 7500
		}
	}

			value = 5000
		}

			# Audax Validator "." Ignore_NEXT
			offmap = ROOT
			value = 2500
		}
	}
}
