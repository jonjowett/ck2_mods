expd_pdxrptg_improve_this_characters_genetics_attractiveness = {
	
	if = {
		limit = { trait = inbred }
		remove_trait = inbred
	}
	else_if = {
		limit = { trait = ugly }
		remove_trait = ugly
	}
	else_if = {
		limit = { trait = uncouth }
		remove_trait = uncouth
	}
	else_if = {
		limit = { NOR = { trait = groomed trait = fair } }
		add_trait = groomed
	}
	else_if = {
		limit = { trait = groomed NOT = { trait = fair } }
		add_trait = fair
	}
	else = {
		# Maxed-out - do nothing
	}
	
}
