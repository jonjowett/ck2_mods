# IMPORTANT NOTE!
#
# These "multiplicative" modifiers are intended for use in a "Paradox-math" style of calculation, ie:
#    {Final cost} = {Base cost} * (1 + {Sum of all "multiplicative" modifiers} )
# Eg:
#  --- For no effect, the "multiplicative" modifier should be 0
#  --- For +50%     , the "multiplicative" modifier should be 0.5
#  --- For +100%    , the "multiplicative" modifier should be 1
#  --- etc

#              INTERCONTINENTAL TRANSPORTATION              #

# If an animal is native to a completely different continent, bringing it to your capital for purchase is probably quite expensive.

# Thought process about animal sizes:
#  --- Tiny   = Easy to hold in hand
#  --- Small  = Average-size dog
#  --- Medium = Average-size mount
#  --- Large  = Unusually large mount
#  --- Huge   = Elephant?!
# Then, upgrade/downgrade one level if an animal is unusually hard/easy to transport.

expd_pdxrptg_purchase_mount_pet_clear_continent_cost_multipliers = {
	set_variable = { which = mount_pet_tamriel_cost_tiny_mult            value = 0 }
	set_variable = { which = mount_pet_tamriel_cost_small_mult           value = 0 }
	set_variable = { which = mount_pet_tamriel_cost_medium_mult          value = 0 }
	set_variable = { which = mount_pet_tamriel_cost_large_mult           value = 0 }
	set_variable = { which = mount_pet_tamriel_cost_huge_mult            value = 0 }
	set_variable = { which = mount_pet_tamriel_yokuda_cost_tiny_mult     value = 0 }
	set_variable = { which = mount_pet_tamriel_yokuda_cost_small_mult    value = 0 }
	set_variable = { which = mount_pet_tamriel_yokuda_cost_medium_mult   value = 0 }
	set_variable = { which = mount_pet_tamriel_yokuda_cost_large_mult    value = 0 }
	set_variable = { which = mount_pet_tamriel_yokuda_cost_huge_mult     value = 0 }
	set_variable = { which = mount_pet_akavir_cost_tiny_mult             value = 0 }
	set_variable = { which = mount_pet_akavir_cost_small_mult            value = 0 }
	set_variable = { which = mount_pet_akavir_cost_medium_mult           value = 0 }
	set_variable = { which = mount_pet_akavir_cost_large_mult            value = 0 }
	set_variable = { which = mount_pet_akavir_cost_huge_mult             value = 0 }
}
