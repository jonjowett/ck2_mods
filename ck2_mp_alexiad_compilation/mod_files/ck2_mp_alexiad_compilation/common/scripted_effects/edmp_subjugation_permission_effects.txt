edmp_remove_subjugation_permission_flag_from_this_character = {
	
	# Show the "all permissions revoked" effect BEFORE the (possibly-very-long) list of individual CBs lost
	custom_tooltip = { text = edmp_remove_subjugation_permission_flag_from_this_character_tt }
	
	# Tell the decision-taker that a notification will be sent.
	# NB: Be sure to keep this limit in-sync with the one below!
	if = {
		limit = {
			edmp_this_is_a_multiplayer_game = yes
			ai = no
		}
		custom_tooltip = { text = edmp_subjugation_permission_notification_sent_tt }
	}
	
	# List of individual CBs lost.
	# Must remove these BEFORE removing the character modifier, because we need the modifier to detect which titles need to have their permissions revoked.
	# Do NOT notify players about each individual permission withdrawn, to avoid spam. (We will send a blanket notification at the end of this event.)
	any_character = {
		limit = { edmp_prev_character_has_permission_to_subjugate_this_character = yes }
		edmp_remove_prev_characters_subjugation_target_flag_from_this_character_without_notification = yes
	}
	
	# Actual "all permissions revoked" effect, hidden because it's got a custom tooltip above.
	hidden_tooltip = {
		remove_character_modifier = edmp_subjugation_permission_character_01
		remove_character_modifier = edmp_subjugation_permission_character_02
		remove_character_modifier = edmp_subjugation_permission_character_03
		remove_character_modifier = edmp_subjugation_permission_character_04
		remove_character_modifier = edmp_subjugation_permission_character_05
		remove_character_modifier = edmp_subjugation_permission_character_06
		remove_character_modifier = edmp_subjugation_permission_character_07
		remove_character_modifier = edmp_subjugation_permission_character_08
		remove_character_modifier = edmp_subjugation_permission_character_09
		remove_character_modifier = edmp_subjugation_permission_character_10
		remove_character_modifier = edmp_subjugation_permission_character_11
		remove_character_modifier = edmp_subjugation_permission_character_12
		remove_character_modifier = edmp_subjugation_permission_character_13
		remove_character_modifier = edmp_subjugation_permission_character_14
		remove_character_modifier = edmp_subjugation_permission_character_15
		remove_character_modifier = edmp_subjugation_permission_character_16
		remove_character_modifier = edmp_subjugation_permission_character_17
		remove_character_modifier = edmp_subjugation_permission_character_18
		remove_character_modifier = edmp_subjugation_permission_character_19
		remove_character_modifier = edmp_subjugation_permission_character_20
	}
	
	# If THIS is a player character in multiplayer, notify all players that the host has withdrawn subjugation permissions.
	# NB: Be sure to keep this limit in-sync with the one above!
	hidden_tooltip = {
		if = {
			limit = {
				edmp_this_is_a_multiplayer_game = yes
				ai = no
			}
			save_event_target_as = edmp_spn_character
			random_player = {
				preferred_limit = { is_multiplayer_host_character = yes }
				any_player = { character_event = { id = edmp_spn.3000 } }
			}
			clear_event_target = edmp_spn_character
		}
	}
	
}
